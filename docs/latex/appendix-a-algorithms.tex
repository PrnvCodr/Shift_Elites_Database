This appendix provides pseudocode for the core algorithms implemented in \shiftdb{} and the database theory concepts covered in this guide.

% ─────────────────── STORAGE ───────────────────
\section{Storage Engine Algorithms}

\begin{algorithm}[H]
\caption{Slotted-Page Tuple Insertion}
\label{alg:a-insert-tuple}
\KwInput{Page $P$, serialised tuple $t$ of length $|t|$}
\KwOutput{Slot number $s$ or $-1$}
\If{$P.\text{getFreeSpace}() < |t| + \text{sizeof(SlotEntry)}$}{
  \Return{$-1$} \tcp*{Page is full}
}
$s \leftarrow -1$\;
\For{$i = 0$ \textbf{to} $P.\text{numSlots} - 1$}{
  \If{$P.\text{slot}[i].\text{occupied} = \text{false}$}{
    $s \leftarrow i$\;
    \textbf{break}\;
  }
}
\If{$s = -1$}{
  $s \leftarrow P.\text{numSlots}$\;
  $P.\text{numSlots} \leftarrow P.\text{numSlots} + 1$\;
}
$P.\text{tupleDataOffset} \leftarrow P.\text{tupleDataOffset} - |t|$\;
Copy $t$ into $P$ at offset $P.\text{tupleDataOffset}$\;
$P.\text{slot}[s] \leftarrow (\text{offset}: P.\text{tupleDataOffset},\ \text{length}: |t|,\ \text{occupied}: \text{true})$\;
\Return{$s$}\;
\end{algorithm}

\begin{algorithm}[H]
\caption{Row Serialisation (Null Bitmap + Typed Fields)}
\label{alg:a-serialize}
\KwInput{Row $R = [v_0, \ldots, v_{n-1}]$, schema $S$ with column types}
\KwOutput{Byte buffer $B$}
$B \leftarrow$ allocate $\lceil n/8 \rceil$ zero bytes \tcp*{Null bitmap}
\For{$i = 0$ \textbf{to} $n - 1$}{
  \If{$v_i$ is NULL}{
    Set bit $i$ in $B$\;
    \textbf{continue}\;
  }
  \Switch{$S.\text{columns}[i].\text{type}$}{
    \Case{INT}{Append 4 bytes (little-endian int32)}
    \Case{FLOAT}{Append 4 bytes (IEEE 754 float)}
    \Case{BOOLEAN}{Append 1 byte (0 or 1)}
    \Case{VARCHAR, DATE, TIMESTAMP}{
      Append 2-byte length prefix $|v_i|$\;
      Append $|v_i|$ raw characters\;
    }
  }
}
\Return{$B$}\;
\end{algorithm}

% ─────────────────── BUFFER POOL ───────────────────
\section{Buffer Pool Algorithms}

\begin{algorithm}[H]
\caption{Buffer Pool --- Fetch Page}
\label{alg:a-fetch}
\KwInput{Table name $T$, page ID $p$}
\KwOutput{Pointer to buffered page}
Lock the buffer pool mutex\;
$\text{key} \leftarrow T + \text{``:''}  + p$\;
\If{key $\in$ pageMap}{
  Increment $\text{frame.pinCount}$\;
  Move key to front of LRU list\;
  Increment $\text{hits}$\;
  \Return{frame.page}\;
}
Increment $\text{misses}$\;
\If{$|\text{pageMap}| \geq \text{maxSize}$}{
  Call \textsc{EvictPage}()\;
}
Create new BufferFrame $f$, set $f.\text{pinCount} \leftarrow 1$\;
Call DiskManager.readPage($T$, $p$, $f.\text{page}$)\;
Insert $f$ into pageMap under key\;
Push key to front of LRU list\;
\Return{$f.\text{page}$}\;
\end{algorithm}

\begin{algorithm}[H]
\caption{LRU Eviction}
\label{alg:a-evict}
\KwOutput{One unpinned page evicted from the buffer pool}
\ForEach{key from LRU tail (least recently used)}{
  $f \leftarrow \text{pageMap}[\text{key}]$\;
  \If{$f.\text{pinCount} = 0$}{
    \If{$f.\text{dirty}$}{
      DiskManager.writePage($f.\text{tableName}$, $f.\text{pageId}$, $f.\text{page}$)\;
    }
    Remove key from pageMap and LRU list\;
    \Return{}\;
  }
}
\tcp{All pages pinned --- pool exhausted}
\end{algorithm}

% ─────────────────── PARSER ───────────────────
\section{Parser Algorithms}

\begin{algorithm}[H]
\caption{Recursive-Descent Expression Parsing}
\label{alg:a-expr}
\SetKwFunction{FParseOr}{parseOr}
\SetKwFunction{FParseAnd}{parseAnd}
\SetKwFunction{FParseCmp}{parseComparison}
\SetKwFunction{FParseAdd}{parseAddition}
\SetKwFunction{FParseMul}{parseMultiplication}
\SetKwFunction{FParsePri}{parsePrimary}

\SetKwProg{Fn}{Function}{}{}
\Fn{\FParseOr{}}{
  $\text{left} \leftarrow$ \FParseAnd{}\;
  \While{current token is \code{OR}}{
    consume OR\;
    $\text{right} \leftarrow$ \FParseAnd{}\;
    $\text{left} \leftarrow \text{BinaryOp(OR, left, right)}$\;
  }
  \Return{left}\;
}
\Fn{\FParseAnd{}}{
  $\text{left} \leftarrow$ \FParseCmp{}\;
  \While{current token is \code{AND}}{
    consume AND\;
    $\text{right} \leftarrow$ \FParseCmp{}\;
    $\text{left} \leftarrow \text{BinaryOp(AND, left, right)}$\;
  }
  \Return{left}\;
}
\tcp{parseComparison, parseAddition, parseMultiplication follow the same pattern at decreasing precedence}
\Fn{\FParsePri{}}{
  \uIf{token is INTEGER}{
    \Return{Literal(parseInt(token))}\;
  }
  \uElseIf{token is STRING}{
    \Return{Literal(token.value)}\;
  }
  \uElseIf{token is IDENTIFIER}{
    Read name; if ``\code{.}'' follows, read qualified column ref\;
    \Return{ColumnRef(name)}\;
  }
  \uElseIf{token is \code{(}}{
    consume \code{(}\; $e \leftarrow$ \FParseOr{}\; consume \code{)}\;
    \Return{$e$}\;
  }
}
\end{algorithm}

% ─────────────────── B-TREE ───────────────────
\section{B-Tree Algorithms}

\begin{algorithm}[H]
\caption{B-Tree Search}
\label{alg:a-btree-search}
\KwInput{Root node $r$, search key $k$}
\KwOutput{Record pointer or \texttt{null}}
$n \leftarrow r$\;
\While{$n$ is not a leaf}{
  $i \leftarrow$ binary search for smallest key $\geq k$ in $n$\;
  \If{$n.\text{key}[i] = k$}{
    \Return{$n.\text{pointer}[i]$}\;
  }
  $n \leftarrow n.\text{child}[i]$\;
}
Search leaf $n$ for $k$\;
\Return{pointer if found, else \texttt{null}}\;
\end{algorithm}

\begin{algorithm}[H]
\caption{B-Tree Insertion}
\label{alg:a-btree-insert}
\KwInput{Key $k$, value $v$}
Find leaf node $L$ where $k$ belongs\;
\If{$L$ has room}{
  Insert $(k, v)$ into $L$ in sorted order\;
}{
  Split $L$ into $L'$ (lower half) and $L''$ (upper half)\;
  $m \leftarrow$ median key\;
  Push $m$ up to parent of $L$\;
  \If{parent overflows}{
    Recursively split parent\;
  }
}
\end{algorithm}

\begin{algorithm}[H]
\caption{B-Tree Deletion}
\label{alg:a-btree-delete}
\KwInput{Key $k$}
Find leaf $L$ containing $k$; remove $k$\;
\If{$|L| < \lceil m/2 \rceil - 1$}{
  \uIf{sibling has extra keys}{
    Redistribute: borrow key through parent\;
  }
  \Else{
    Merge $L$ with sibling; pull separator from parent\;
    \If{parent underflows}{Recursively fix\;}
  }
}
\end{algorithm}

% ─────────────────── ARIES ───────────────────
\section{Recovery Algorithms}

\begin{algorithm}[H]
\caption{ARIES --- Full Recovery}
\label{alg:a-aries}
\SetKwFunction{FAnalysis}{Analysis}
\SetKwFunction{FRedo}{Redo}
\SetKwFunction{FUndo}{Undo}
\tcp{Phase 1: Analysis}
$\text{DPT} \leftarrow \emptyset$; $\text{ATT} \leftarrow \emptyset$\;
\ForEach{log record $r$ from last checkpoint (forward)}{
  \If{$r$ is data modification}{
    Add $r.\text{pageId}$ to DPT (if absent) with $r.\text{LSN}$ as recLSN\;
    Mark $r.\text{txnId}$ active in ATT\;
  }
  \If{$r$ is COMMIT}{Remove $r.\text{txnId}$ from ATT\;}
}
\tcp{Phase 2: Redo}
$\text{startLSN} \leftarrow \min(\text{recLSN values in DPT})$\;
\ForEach{log record $r$ from startLSN (forward)}{
  \If{$r.\text{pageId} \in \text{DPT}$ \textbf{and} $r.\text{LSN} \geq \text{DPT}[r.\text{pageId}].\text{recLSN}$}{
    Redo the modification described by $r$\;
  }
}
\tcp{Phase 3: Undo}
\ForEach{transaction $T$ still in ATT (loser transactions)}{
  \ForEach{log record of $T$ in reverse LSN order}{
    Undo the modification; write CLR\;
  }
}
\end{algorithm}

% ─────────────────── QUERY ───────────────────
\section{Query Execution Algorithms}

\begin{algorithm}[H]
\caption{Expression Evaluation}
\label{alg:a-eval-expr}
\KwInput{AST node $e$, row context $\text{ctx}$}
\KwOutput{Value $v$}
\Switch{$e.\text{type}$}{
  \Case{LITERAL}{\Return{$e.\text{value}$}\;}
  \Case{COLUMN\_REF}{
    \Return{$\text{ctx}[e.\text{columnName}]$}\;
  }
  \Case{BINARY\_OP}{
    $l \leftarrow \text{evaluate}(e.\text{left}, \text{ctx})$\;
    $r \leftarrow \text{evaluate}(e.\text{right}, \text{ctx})$\;
    \Return{apply $e.\text{op}$ to $(l, r)$}\;
  }
  \Case{AGGREGATE}{
    \Return{compute aggregate over group}\;
  }
}
\end{algorithm}

\begin{algorithm}[H]
\caption{Hash Join (Not Yet Implemented)}
\label{alg:a-hash-join}
\KwInput{Left table $L$, right table $R$, join key columns}
\KwOutput{Joined rows}
\tcp{Build phase}
$H \leftarrow$ empty hash table\;
\ForEach{row $r \in R$ (smaller table)}{
  $k \leftarrow$ extract join key from $r$\;
  $H[k].\text{append}(r)$\;
}
\tcp{Probe phase}
\ForEach{row $l \in L$}{
  $k \leftarrow$ extract join key from $l$\;
  \ForEach{row $r \in H[k]$}{
    emit combined row $(l, r)$\;
  }
}
\end{algorithm}
